A simple placeholder animation component used to indicate content loading.

<ComponentPreview name="skeleton" />

---

## Installation

To add the `Skeleton` to your project:

<Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>
  <Tab value="npm">
    ```bash
    npx @gravcn/cli add skeleton
    ```
  </Tab>
  <Tab value="pnpm">
    ```bash
    pnpm dlx @gravcn/cli add skeleton
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn dlx @gravcn/cli add skeleton
    ```
  </Tab>
  <Tab value="bun">
    ```bash
    bunx @gravcn/cli add skeleton
    ```
  </Tab>
</Tabs>


---

## Usage Example

Here is how you can render the `Skeleton` component:

```tsx title="demo/Skeleton.tsx"
import { Skeleton } from "@/components/ui/skeleton";

export function SkeletonDemo() {
  return (
    <div className="flex items-center space-x-4">
      <Skeleton className="h-12 w-12 rounded-full" />
      <div className="space-y-2">
        <Skeleton className="h-4 w-[250px]" />
        <Skeleton className="h-4 w-[200px]" />
      </div>
    </div>
  );
}
```
